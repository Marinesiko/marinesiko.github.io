           import requests

class Delete_new_location():
    """"Удаление новых локаций"""

    def test_delete_location(self):

        """"Отправить метод DELETE,  с помощью которого удалить 2-й и 4-й place_id из текстового файла"""

        base_url = "https://rahulshettyacademy.com"  # базовый url
        delete_resource = "/maps/api/place/delete/json"
        key = "?key=qaclick123"  # параметр для всех запросов
        delete_url = base_url + delete_resource + key
        file = open("place_id_google.txt", "r")
        content = file.readlines()

        spisok = [1, 3]
        print("***Отправить метод DELETE,  с помощью которого удалить 2-й и 4-й place_id из текстового файла***")
        for i in spisok:
            place_id = content[i].rstrip()
            json_for_delete_new_location = {
                "place_id": place_id
            }
            result_delete = requests.delete(delete_url, json=json_for_delete_new_location)
            print(result_delete.text)
            print("Статус код : " + str(result_delete.status_code))
            assert 200 == result_delete.status_code
            if result_delete.status_code == 200:
                print("Удаление локации прошло успешно")
            else:
                print("Провал!!! Запрос ошибочный")

        """Проверка удаления новой локации"""

        base_url_get = "https://rahulshettyacademy.com"  # базовый url
        get_resource_get = "/maps/api/place/get/json"

        print("***Отправить метод Get который будет читать place_id из текстового файла, и сделает отбор на существующие и несуществующие локации***")

        with open("place_id_google.txt", "r") as file:
            for i in file:

                last_line_string = i.rstrip()
                place_id_line = str("&place_id=" + last_line_string)
                get_url = base_url_get + get_resource_get + key + place_id_line
                result_get = requests.get(get_url)
                print("Place_id : " + last_line_string + " Статус код : " + str(result_get.status_code))
                if result_get.status_code == 200:
                    """Создаем файл с существующими локациями"""
                    print("Локация существует")
                    file_for_place_id = open('place_good_id.txt', 'a')  # открытие в режиме записи
                    file_for_place_id.write(last_line_string + '\n')
                else:
                    print("Данная локация не существует")


        print("Тестирование Delete_new_location завершено успешно")
